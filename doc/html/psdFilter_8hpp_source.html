22 /// A class for filtering noise with PSDs
23 /** The square-root of the PSD is maintained by this class, either as a pointer to an external array or using internally allocated memory (which will be
24  * de-allocated on destruction.
25  *
26  * PSD Normalization: the PSD used for this needs to be normalized properly to produce filtered noise with the correct statistics. Given an
27  * array of size N which contains the PSD as "power per unit frequency" vs frequency (i.e. the PSD) on some frequency scale with uniform spacing \f$ \Delta f \f$,
28  * the normalization is
29  \f[
30  PSD_{norm} = PSD * N * \Delta f
31  \f]
32  * for a 1-dimensional PSD and
33  \f[
34  PSD_{norm} = PSD * N_0 * \Delta f_0 * N_1 * \Delta f_1
35  \f]
36  * for a 2-dimensional PSD. Remember that these are applied before taking the square root.
37  *
38  *
39  * \ingroup psds
40  *
41  * \todo once fftT has a plan interface with pointers for working memory, use it.
42  */

277 template<typename realT>
278 int psdFilter<realT>::filter( realArrayT & noise )
279 {
280  //Make noise a complex number
281  for(int ii=0;ii<noise.rows();++ii)
282  {
283  for(int jj=0; jj<noise.cols(); ++jj)
284  {
285  _ftWork(ii,jj) = complexT(noise(ii,jj),0);
286  }
287  }
288 
289  //Transform complex noise to Fourier domain.
290  fft_fwd(_ftWork.data(), _ftWork.data() );
291 
292  //Apply the filter.
293  _ftWork *= *_psdSqrt;
294 
295  fft_back(_ftWork.data(), _ftWork.data());
296 
297  //Now take the real part, and normalize.
298  noise = _ftWork.real()/(noise.rows()*noise.cols());
299 
300  return 0;
301 }

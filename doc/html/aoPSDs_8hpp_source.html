190  ///Get the value of the PSD at spatial frequency k and a zenith distance.
191  /**
192  *
193  * \returns the von Karman PSD at the specified spatial frequency.
194  * \returns -1 if an error occurs.
195  *
196  */
197  realT operator()( aoAtmosphere<realT> & atm, ///< [in] gives the atmosphere parameters r_0 and L_0.
198  realT k, ///< [in] is the spatial frequency in m^-1.
199  int n, //place holder
200  realT sec_zeta ///< [in] is the secant of the zenith distance.
201  )
202  {
203  realT k02;
204 
205  if(atm.L_0() > 0)
206  {
207  k02 = (1)/(atm.L_0()*atm.L_0());
208  }
209  else k02 = 0;
210 
211  if(k02 == 0 && k == 0)
212  {
213  return 0;
214  }
215 
216  realT Ppiston, Ptiptilt;
217 
218  if( (_subPiston || _subTipTilt) )
219  {
220  if (_D == 0)
221  {
222  mxError("aoAtmosphere", MXE_PARAMNOTSET, "Diameter D not set for Piston and/or TT subtraction.");
223  return -1;
224  }
225 
226  if(_subPiston)
227  {
228  Ppiston = pow(2*math::func::jinc(pi<realT>()*k*_D), 2);
229  }
230  else Ppiston = 0;
231 
232  if(_subTipTilt)
233  {
234  Ptiptilt = pow(4*math::func::jinc2(pi<realT>()*k*_D), 2);
235  }
236  else Ptiptilt = 0;
237  }
238  else
239  {
240  Ppiston = 0;
241  Ptiptilt = 0;
242  }
243 
244  return constants::a_PSD<realT>()*pow(atm.r_0(), -five_thirds<realT>())*pow(k*k+k02, -eleven_sixths<realT>()) * (1.0-Ppiston - Ptiptilt)*sec_zeta;
245  }

247  /// Get the value of the PSD at spatial frequency k and wavelength lambda, and a zenith distance, with a WFS at a different wavelength
248  /**
249  *
250  * \returns the von Karman PSD at the specified spatial frequency for the specified wavelength.
251  * \returns -1 if an error occurs.
252  */
253  realT operator()( aoAtmosphere<realT> & atm, ///< [in] gives the atmosphere parameters r_0 and L_0.
254  realT k, ///< [in] is the spatial frequency in m^-1.
255  realT lambda, ///< [in] is the observation wavelength in m
256  int n, //place holder
257  realT lambda_wfs, ///< [in] is the wavefront measurement wavelength in m
258  realT secZeta ///< [in] is the secant of the zenith distance
259  )
260  {
261  realT psd = operator()(atm, k, 0, secZeta)* pow( atm.lam_0()/lambda, 2);
262 
263  if(psd < 0) return -1;
264 
265  if(_scintillation)
266  {
267  if(_component == PSDComponent::phase)
268  {
269  psd *= atm.X(k, lambda, secZeta);
270  }
271  else if (_component == PSDComponent::amplitude)
272  {
273  psd *= atm.Y(k, lambda, secZeta);
274  }
275  else if (_component == PSDComponent::dispPhase)
276  {
277  psd *= atm.X_Z(k, lambda, lambda_wfs, secZeta);
278  }
279  else if (_component == PSDComponent::dispAmplitude)
280  {
281  mxError("vonKarmanSpectrum::operator()", MXE_NOTIMPL, "Dispersive-aniso amplitude not implemented");
282  return 0;
283  }
284  else
285  {
286  mxError("vonKarmanSpectrum::operator()", MXE_INVALIDARG, "Invalid component specified");
287  return 0;
288  }
289  }
290 
291  return psd;
292 
293  }

410 template<typename realT, typename aosysT>
411 realT fourierTemporalPSD<realT, aosysT>::fastestPeak( int m,
412  int n )
413 {
414  realT ku, kv, vu,vv;
415 
416  ku = ( (realT) m / _aosys->D());
417  kv = ( (realT) n / _aosys->D());
418 
419  realT f, fmax = 0;
420 
421  for(int i=0; i< _aosys->atm.n_layers(); ++i)
422  {
423  vu = _aosys->atm.layer_v_wind(i) * cos(_aosys->atm.layer_dir(i));
424  vv = _aosys->atm.layer_v_wind(i) * sin(_aosys->atm.layer_dir(i));
425 
426  f = fabs(ku*vu + kv*vv);
427  if(f > fmax) fmax = f;
428  }
429 
430  return fmax;
431 }

433 template<typename realT, typename aosysT>
434 int fourierTemporalPSD<realT, aosysT>::singleLayerPSD( std::vector<realT> &PSD,
435  std::vector<realT> &freq,
436  realT m,
437  realT n,
438  int layer_i,
439  int p,
440  realT fmax )
441 {
442 
443 
444  if(fmax == 0) fmax = freq[freq.size()-1];
445 
446  realT v_wind = _aosys->atm.layer_v_wind(layer_i);
447  realT q_wind = _aosys->atm.layer_dir(layer_i);
448  realT r_0 = _aosys->atm.r_0();
449  realT D = _aosys->D();
450 
451  //Rotate the basis
452  realT cq = cos(q_wind);
453  realT sq = sin(q_wind);
454 
455 
456  realT scale = 2*(1/v_wind); //Factor of 2 for negative frequencies.
457 
458  //We'll get the occasional failure to reach tolerance error, just ignore them all for now.
459  gsl_set_error_handler_off();
460 
461  //Create a local instance so that we're reentrant
462  fourierTemporalPSD<realT, aosysT> params(true);
463 
464  params._aosys = _aosys;
465  params._layer_i = layer_i;
466  params._m = m*cq + n*sq;
467  params._n = -m*sq + n*cq;
468  params._p = p;
469 
470 
471  params._mode_i = _mode_i;
472  params._modeCoeffs = _modeCoeffs;
473 
474 
475  realT result, error, result2;
476 
477  //Setup the GSL calculation
478  gsl_function func;
479  switch(_useBasis)
480  {
481  case basis::basic: //MXAO_FTPSD_BASIS_BASIC:
482  func.function = &F_basic<realT, aosysT>;
483  break;
484  case basis::modified: //MXAO_FTPSD_BASIS_MODIFIED:
485  func.function = &F_mod<realT, aosysT>;
486  break;
487  case basis::projected_basic: //MXAO_FTPSD_BASIS_PROJECTED_BASIC:
488  mxError("fourierTemporalPSD::singleLayerPSD", MXE_NOTIMPL, "Projected basic-basis modes are not implemented.");
489  break;
490  case basis::projected_modified:// MXAO_FTPSD_BASIS_PROJECTED_MODIFIED:
491  params.Jps = Jps;
492  params.Jms = Jms;
493  params.ms = ms;
494  params.ns = ns;
495  params.ps = ps;
496 
497 
498  func.function = &F_projMod<realT, aosysT>;
499 
500  break;
501  default:
502  mxError("fourierTemporalPSD::singleLayerPSD", MXE_INVALIDARG, "value of _useBasis isn not valid.");
503  return -1;
504  }
505 
506  func.params = &params;
507 
508 
509  //Here we only calculate up to fmax.
510  int i=0;
511  while(freq[i] <= fmax && i < freq.size())
512  {
513  params._f = freq[i];
514 
515  int ec = gsl_integration_qagi (&func, _absTol, _relTol, WSZ, params._w, &result, &error);
516 
517  if(ec == GSL_EDIVERGE)
518  {
519  std::cerr << "GSL_EDIVERGE:" << p << " " << freq[i] << " " << v_wind << " " << m << " " << n << " " << _m << " " << _n << "\n";
520  std::cerr << "ignoring . . .\n";
521  }
522 
523  PSD[i] = scale*result;
524 
525  ++i;
526  }
527 
528  //Now fill in from fmax to the actual max frequency with a -17/3 power law.
529  int j=i;
530 
531  //First average result for last 50.
532  PSD[j] = PSD[i-50] * pow( freq[i-50]/freq[j], seventeen_thirds<realT>());
533  for(int k=49; k> 0; --k)
534  {
535  PSD[j] += PSD[i-k] * pow( freq[i-k]/freq[j], seventeen_thirds<realT>());
536  }
537  PSD[j] /= 50.0;
538  ++j;
539  ++i;
540  while(j < freq.size())
541  {
542  PSD[j] = PSD[i-1] * pow( freq[i-1]/freq[j], seventeen_thirds<realT>());
543  ++j;
544  }
545 
546 
547 
548  return 0;
549 }

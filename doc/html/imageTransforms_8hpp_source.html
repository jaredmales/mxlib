181  Ncols = im.cols();
182 
183

185

188  ycen = 0.5*(Ncols-1.);

191  int yulim = Ncols-width+lbuff;// - 1;

194  arithT xc_x_sinq = xcen*sinq;
195  arithT yc_x_cosq = ycen*cosq;
196  arithT yc_x_sinq = ycen*sinq;
197 
198  xc_x_cosq += yc_x_sinq;
199  xc_x_sinq -= yc_x_cosq;
200 
201 
202  #pragma omp parallel private(x0,y0,i0,j0,x,y)
203  {

207 
208  #pragma omp for schedule(static, 1)

210  {
211  i_x_cosq = i*cosq - xc_x_cosq;// + xcen;
212  i_x_sinq = -(i*sinq - xc_x_sinq);// + ycen;
213

392  *

490 /// Down-sample an image, reducing its size while conserving the total flux.
491 /** If the old size is an integer multiple of the new size, this is just a re-bin. If not an integer multiple,
492  * the image is interpolated after performing the closest re-bin, and then re-normalized to conserve flux.
493  *
494  * \todo Allow selection of interpolator, providing a default version.
495  */

497 void imageDownSample(imageOutT & imout, const imageInT & imin)
498 {

501

503  Scalar inputTotal = fabs(imin.sum());
504 
505 
506 
507  //As a first step, rebin to nearest whole pixel factor which is larger than the desired output size
508  int closestRebin = imin.rows()/imout.rows();//, imin.cols()/imout.cols() );
509 
510  float sample = ( (float) imin.rows())/ closestRebin;
511 
512  while( sample != floor(sample))
513  {
514  --closestRebin;
515  if(closestRebin == 1) break;
516  sample = ( (float) imin.rows())/ closestRebin;
517  }
518

524  {
525  for(int j=0; j<temp.cols(); ++j)
526  {
527  temp(i,j) = imin.block( i*closestRebin, j*closestRebin, closestRebin, closestRebin).sum();
528  }
529  }
530 
531  //If the output image is now the requested size return.
532  if(temp.rows() == imout.rows() && temp.cols() == imout.cols())
533  {
534  imout = temp;
535  Scalar outputTotal = fabs(imout.sum());
536 
537  //Normalize

540  }

546  imageOutT kern;
547

549  const int width = transformT::width;

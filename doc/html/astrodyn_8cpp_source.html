155 void calc_AZ_EL(double *az, double *el, double ha, double dec, double lat)
156 {
157  double ha_rad, dec_rad, lat_rad, az_rad, el_rad;
158 
159  //printf("%f\n", get_ha()*15.0);
160  ha_rad = ha*15.0*DPI/180.0;
161  dec_rad = dec*DPI/180.0;
162  lat_rad = lat*DPI/180.0;
163 
164  az_rad = atan2(sin(ha_rad), cos(ha_rad)*sin(lat_rad)-tan(dec_rad)*cos(lat_rad));
165 
166  el_rad = asin(sin(lat_rad)*sin(dec_rad) + cos(lat_rad)*cos(dec_rad)*cos(ha_rad));
167 
168  *az = az_rad*180.0/DPI + 180.0;
169  *el = el_rad*180.0/DPI;
170 
171  return;

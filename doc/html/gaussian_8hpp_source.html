138 ///Find value at position (x,y) of the 2D general elliptical Gaussian
139 /**
140  * Computes:
141  *
142  * \f$ G(x,y) = G_0 + A\exp [-0.5( a(x-x_0)^2 + b(x-x_0)(y-y_0) + c(y-y_0)^2 ]\f$
143  *
144  * where, for a counter-clockwise rotation \f$ \theta \f$, we have
145  *
146  * \f$ a = \frac{\cos^2 \theta}{\sigma_x^2} + \frac{\sin^2\theta}{\sigma_y^2}\f$
147  *
148  * \f$ b = \frac{\sin 2\theta}{2} \left(\frac{1}{\sigma_x^2} - \frac{1}{\sigma_y^2} \right)\f$
149  *
150  * \f$ c = \frac{\sin^2 \theta}{\sigma_x^2} + \frac{\cos^2\theta}{\sigma_y^2}\f$
151  *
152  * In this version the parameters are specified directly as (a,b,c), in particular avoiding the trig function calls.
153  * This should be much more efficient, and so this version should be used inside fitting routines, etc. However
154  * note that the matrix {a b}{b c} must be <a href="http://mathworld.wolfram.com/PositiveDefiniteMatrix.html">positive-definite</a>
155  * otherwise infinities can result from the argument of the exponent being positive.
156  *
157  * The functions \ref gaussian2D_gen2rot and \ref gaussian2D_rot2gen provide conversions from
158  * (a,b,c) to (\f$\sigma_x\f$,\f$\sigma_y\f$,\f$\theta\f$) and back. The function \ref gaussian2D_ang is a
159  * wrapper for this function, which instead accepts (\f$\sigma_x\f$,\f$\sigma_y\f$,\f$\theta\f$) as inputs.
160  *
161  * \param x is the x-position at which to evaluate the Gaussian
162  * \param y is the y-positoin at which to evaluate the Gaussian
163  * \param G0 is the constant to add to the Gaussian
164  * \param A is the scaling factor (peak = A)
165  * \param x0 is the x-coordinate of the center
166  * \param y0 is the y-coordinate of the center
167  * \param a is the first rotation and scaling factor
168  * \param b is the second rotation and scaling factor
169  * \param c is the third rotation and scaling factor
170  *
171  * \returns the value of the 2D elliptical Gaussian at (x,y)
172  *
173  * \tparam realT is type to use for arithmetic
174  */
175 template<typename realT>
176 realT gaussian2D( const realT & x,
177  const realT & y,
178  const realT & G0,
179  const realT & A,
180  const realT & x0,
181  const realT & y0,
182  const realT & a,
183  const realT & b,
184  const realT & c )
185 {
186  realT dx = x-x0;
187  realT dy = y-y0;
188 
189  realT ans;
190  ans = G0 + A*std::exp( -0.5* ( a*dx*dx + 2.*b*dx*dy + c*dy*dy));
191 
192  return ans;
193 }

195 ///Convert from (a,b,c) to (\f$\sigma_x\f$,\f$\sigma_y\f$,\f$\theta\f$) for the elliptical Gaussian.
196 /** The general 2D elliptical Gaussian
197  *
198  * \f$ G(x,y) = G_0 + A\exp [-0.5( a(x-x_0)^2 + b(x-x_0)(y-y_0) + c(y-y_0)^2 ]\f$
199  *
200  * can be expressed in terms of widths and a rotation angle using:
201  *
202  * \f$ a = \frac{\cos^2 \theta}{\sigma_x^2} + \frac{\sin^2\theta}{\sigma_y^2}\f$
203  *
204  * \f$ b = \frac{\sin 2\theta}{2} \left(\frac{1}{\sigma_x^2} - \frac{1}{\sigma_y^2} \right)\f$
205  *
206  * \f$ c = \frac{\sin^2 \theta}{\sigma_x^2} + \frac{\cos^2\theta}{\sigma_y^2}\f$
207  *
208  * where \f$ \theta \f$ specifies a counter-clockwise rotation.
209  *
210  * This function calculates (\f$\sigma_x\f$,\f$\sigma_y\f$,\f$\theta\f$) from inputs (a,b,c). It adopts
211  * the convention that the long axis of the ellipse is \f$\sigma_x\f$, and \f$\theta\f$ is chosen
212  * appropriately. Note that \f$-\frac{\pi}{2} < \theta < \frac{\pi}{2}\f$.
213  *
214  * \param sigma_x [output]
215  * \param sigma_y [output]
216  * \param theta [output]
217  * \param a [input]
218  * \param b [input]
219  * \param c [input]
220  *
221  * \tparam realT is type to use for arithmetic
222  */
223 template<typename realT>
224 void gaussian2D_gen2rot( realT & sigma_x,
225  realT & sigma_y,
226  realT & theta,
227  const realT & a,
228  const realT & b,
229  const realT & c )
230 {
231  realT x1, x2, s, s2, theta0, theta1;
232 
233 
234 
235  realT arg = a*a - 2*a*c + 4*b*b + c*c;
236  if(arg < 0)
237  {
238  x2 = 0.5*(a+c);
239  x1 = x2;
240  }
241  else
242  {
243  //There are two roots, one is 1/sigma_x^2 and one is 1/sigma_y^2
244  x2 = 0.5*(a + c) + 0.5*sqrt(arg);
245  x1 = 0.5*(a + c) - 0.5*sqrt(arg);
246  }
247 
248  //Our convention is that the larger width is sigma_x
249  sigma_x = sqrt(1./x1);
250  sigma_y = sqrt(1./x2);
251 
252  //----------------------------------------------------------------------------//
253  // Choosing theta:
254  // There is an ambiguity in theta and which direction (x,y) is the long axis
255  // Here we choose theta so that it specifies the long direction, sigma_x
256  //----------------------------------------------------------------------------//
257 
258  //s is (sin(theta))^2
259  //s2 = (a-x2)*(a-x2)/(b*b + (a-x2)*(a-x2));
260  s = (a-x1)*(a-x1)/(b*b + (a-x1)*(a-x1));
261 
262  //First check if x1-x2 will be close to zero
263 
264  if(fabs(x1-x2) < 1e-12)
265  {
266  theta = 0;
267  return;
268  }
269 
270  theta0 = 0.5*asin( 2*b/(x1-x2)); //This always gives the correct sign
271  theta1 = asin(sqrt(s)); //This always gives the correct magnitude, and seems to be more accurate
272 
273  //Compare signs. If they match, then we use theta1
274  if( std::signbit(theta0) == std::signbit(theta1))
275  {
276  theta = theta1;
277  }
278  else
279  {
280  //Otherwise, we switch quadrants
281  theta = asin(-sqrt(s));
282  }
283 
284 }

294  *
295  * \f$ b = \frac{\sin 2\theta}{2} \left(\frac{1}{\sigma_x^2} - \frac{1}{\sigma_y^2} \right)\f$
296  *

298  *
299  * where \f$ \theta \f$ specifies a counter-clockwise rotation.
300  *
301  * This function calculates (a,b,c) from inputs (\f$\sigma_x\f$,\f$\sigma_y\f$,\f$\theta\f$).
302  *
303  * \param a [output]
304  * \param b [output]
305  * \param c [output]
306  * \param sigma_x [input]
307  * \param sigma_y [input]
308  * \param theta [input]
309  *
310  * \tparam realT is type to use for arithmetic
311  */
312 template<typename realT>
313 void gaussian2D_rot2gen( realT & a,
314  realT & b,
315  realT & c,
316  const realT & sigma_x,
317  const realT & sigma_y,
318  const realT & theta )
319 {
320  realT sn,cs, sx2, sy2;
321 
322  sn = sin(theta);
323  cs = cos(theta);
324  sx2 = sigma_x * sigma_x;
325  sy2 = sigma_y * sigma_y;
326 
327  a = cs*cs/sx2 + sn*sn/sy2;
328  b = sn*cs*(1./sx2-1./sy2);
329  c = sn*sn/sx2 + cs*cs/sy2;
330 }

509 ///Calculate the Jacobian at position (x,y) for the 2D general elliptical Gaussian
510 /**
511  * Given:
512  *
513  * \f$ G(x,y) = G_0 + A\exp [-0.5( a(x-x_0)^2 + b(x-x_0)(y-y_0) + c(y-y_0)^2 ]\f$
514  *
515  * we calculate
516  *
517  * \f$ \frac{\partial G}{\partial G_0} = 1 \f$
518  *
519  * \f$ \frac{\partial G}{\partial A} = (G - G_0)/A \f$
520  *
521  * \f$ \frac{\partial G}{\partial x_0} = 0.5(G-G_0)( 2a(x-x_0) + b(y-y_0)) \f$
522  *
523  * \f$ \frac{\partial G}{\partial y_0} = 0.5(G-G_0) ( b(x-x_0) + 2c(y-y_0)) \f$
524  *
525  * \f$ \frac{\partial G}{\partial a} = -0.5(G-G_0) ( (x-x_0)^2 ) \f$
526  *
527  * \f$ \frac{\partial G}{\partial b} = -0.5(G-G_0) ( (x-x_0)(y-y_0) ) \f$
528  *
529  * \f$ \frac{\partial G}{\partial c} = -0.5(G-G_0) ( (y-y_0)^2 )\f$
530  *
531  * \param j is a 7 element vector which is populated with the derivatives
532  * \param x is the x-position at which to evaluate the Gaussian
533  * \param y is the y-positoin at which to evaluate the Gaussian
534  * \param G0 is the constant to add to the Gaussian
535  * \param A is the scaling factor (peak = A)
536  * \param x0 is the x-coordinate of the center

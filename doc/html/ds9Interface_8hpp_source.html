271 int ds9Interface::spawn()
272 {
273  int i;
274  char resp[32];
275  char cmd[DS9_CMD_MAX_LENGTH];
276  int ret;
277 
278  snprintf(cmd, DS9_CMD_MAX_LENGTH, "xpaaccess %s", _title.c_str());
279 
280  resp[0] = 0;
281  if( command_response(cmd, resp, 128) ) return -1;
282 
283  //Don't respawn if it already exists.
284  if(strcmp(resp, "yes\n") == 0) return 0;
285 
286  ret = system("xpans &");
287  if (WIFSIGNALED(ret))
288  {
289  if( WTERMSIG(ret) == SIGINT) raise(SIGINT);
290  if( WTERMSIG(ret) == SIGQUIT) raise(SIGQUIT);
291  }
292 
293  snprintf(cmd, DS9_CMD_MAX_LENGTH, "ds9 -title %s -port %i &", _title.c_str(), _port);
294 
295  ret = system(cmd);
296  if (WIFSIGNALED(ret))
297  {
298  if( WTERMSIG(ret) == SIGINT) raise(SIGINT);
299  if( WTERMSIG(ret) == SIGQUIT) raise(SIGQUIT);
300  }
301 
302  //Now wait for ds9 to respond or timeout.
303  snprintf(cmd, DS9_CMD_MAX_LENGTH, "xpaaccess %s", _title.c_str());
304 
305  for(i=0;i<10000;i++)
306  {
307  resp[0] = 0;
308  if( command_response(cmd, resp, 128) ) return -1;
309  if(strcmp(resp, "yes\n") == 0) return 0;
310  usleep(100);
311  }
312 
313  return -1; //a timeout
314 }

360 int ds9Interface::display( const dataT * im,
361  size_t dim1,
362  size_t dim2,
363  size_t dim3,
364  int frame
365  )
366 {
367  size_t i, tot_size;
368  char cmd[DS9_CMD_MAX_LENGTH];
369  size_t pixsz;
370  int bitpix;
371  int ret;
372 
373  if(frame < 1)
374  {
375  fprintf(stderr, "frame must >= 1\n");
376  return -1;
377  }
378 
379  spawn();
380 
381  bitpix = getFitsBITPIX<dataT>();
382 
383  pixsz = sizeof(dataT);
384 
385  //If needed add a shared memory segment for this frame
386  if(frame-1 >= nsegs)
387  {
388  addsegment( frame );
389  }
390 
391  //Calculate total size
392  tot_size= pixsz;
393  tot_size*=dim1;
394  tot_size*=dim2;
395  tot_size*=dim3;
396 
397  //Re-allocate shared memory if necessary
398  if(tot_size > segs[frame-1].size)
399  {
400  if( segs[frame-1].size > 0 )
401  {
402  sharedmem_segment_detach( &segs[frame-1]);
403  }
404  sharedmem_segment_create( &segs[frame-1], tot_size);
405  }
406 
407  memcpy( segs[frame-1].addr, im, tot_size );
408 
409 
410  snprintf(cmd, DS9_CMD_MAX_LENGTH, "xpaset -p %s frame %i", _title.c_str(), frame);
411  ret = system(cmd);
412  if (WIFSIGNALED(ret))
413  {
414  if( WTERMSIG(ret) == SIGINT) raise(SIGINT);
415  if( WTERMSIG(ret) == SIGQUIT) raise(SIGQUIT);
416  }
417 
418 
419  //Handle single image so that the cube dialog doesn't open up if dim3=1
420  if(dim3 == 1)
421  {
422  snprintf(cmd, DS9_CMD_MAX_LENGTH, "xpaset -p %s shm array shmid %i [xdim=%zu,ydim=%zu,bitpix=%i] &",
423  _title.c_str(), segs[frame-1].shmemid,
424  dim1, dim2, bitpix);
425  }
426  else
427  {
428  snprintf(cmd, DS9_CMD_MAX_LENGTH, "xpaset -p %s shm array shmid %i [xdim=%zu,ydim=%zu,zdim=%zu,bitpix=%i] &",
429  _title.c_str(), segs[frame-1].shmemid,
430  dim1, dim2, dim3, bitpix);
431  }
432 
433  ret = system(cmd);
434  if (WIFSIGNALED(ret))
435  {
436  if( WTERMSIG(ret) == SIGINT) raise(SIGINT);
437  if( WTERMSIG(ret) == SIGQUIT) raise(SIGQUIT);

387 inline int fitsHeaderCard::write(fitsfile *fptr)
388 {
389  int fstatus;
390  switch(type)
391  {
392  case TSTRING:
393  {
394  return fits_write_key<char *>(fptr, (char *) keyword.c_str(), (void *)value.c_str(), (char *) comment.c_str());
395  }
396  case TBYTE:
397  {
398  unsigned char b = Int();
399  return fits_write_key<unsigned char>(fptr, (char *) keyword.c_str(), &b, (char *) comment.c_str());
400  }
401  case TSBYTE:
402  {
403  signed char sc = Int();
404  return fits_write_key<signed char>(fptr, (char *) keyword.c_str(), &sc, (char *) comment.c_str());
405  }
406  case TSHORT:
407  {
408  short s = Int();
409  return fits_write_key<short>(fptr, (char *) keyword.c_str(), &s, (char *) comment.c_str());
410  }
411  case TUSHORT:
412  {
413  unsigned short us = Int();
414  return fits_write_key<unsigned short>(fptr, (char *) keyword.c_str(), &us, (char *) comment.c_str());
415  }
416  case TINT:
417  {
418  int i = Int();
419  return fits_write_key<int>(fptr, (char *) keyword.c_str(), &i, (char *) comment.c_str());
420  }
421  case TLONG:
422  {
423  long l = Long();
424  return fits_write_key<long>(fptr, (char *) keyword.c_str(), &l, (char *) comment.c_str());
425  }
426  case TLONGLONG:
427  {
428  long long ll = longLong();
429  return fits_write_key<long long>(fptr, (char *) keyword.c_str(), &ll, (char *) comment.c_str());
430  }
431  case TULONG:
432  {
433  unsigned long ul = unsignedLong();
434  return fits_write_key<unsigned long>(fptr, (char *) keyword.c_str(), &ul, (char *) comment.c_str());
435  }
436  case TFLOAT:
437  {
438  float f = Float();
439  return fits_write_key<float>(fptr, (char *) keyword.c_str(), &f, (char *) comment.c_str());
440  }
441  case TDOUBLE:
442  {
443  double d = Double();
444  return fits_write_key<double>(fptr, (char *) keyword.c_str(), &d, (char *) comment.c_str());
445  }
446  case fitsTCOMMENT:
447  {
448  return fits_write_comment(fptr, (char *) comment.c_str());
449  }
450  case fitsTHISTORY:
451  {
452  return fits_write_history(fptr, (char *) comment.c_str());
453  }
454  default:
455  {
456  return fits_write_key<fitsUnknownType>(fptr, (char *) keyword.c_str(), (void *)value.c_str(), (char *) comment.c_str());
457  }
458  }

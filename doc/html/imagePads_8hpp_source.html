89 ///Pad an image by repeating the values in the edge rows and columns
90 /** Allocates imOut to hold 2*padSz more rows and columns, and copies imIn to the center. Then fills in the new pixels
91  * by copying the edge pixels outward.
92  *
93  * \param [out] imOut is resized, and on return contains the padded image.
94  * \param [in] imIn the image to be padded.
95  * \param [in] padSz the size of the pad to be added. The image will grow by 2*padSz rows and columns.
96  *
97  * \retval 0 on success
98  * \retval -1 on error
99  *
100  * \tparam imOutT is an Eigen-like array
101  * \tparam imInT is an Eigen-like array
102  */
103 template<typename imOutT, typename imInT>
104 int padImage( imOutT & imOut,
105  imInT & imIn,
106  unsigned int padSz )
107 {
108  int dim1 = imIn.rows();
109  int dim2 = imIn.cols();
110 
111  imOut.resize(dim1+2*padSz, dim2 + 2*padSz);
112 
113  imOut.block(padSz, padSz, dim1, dim2) = imIn;
114 
115  //First do the left and right edge columns and corners
116  for(int i = 0; i< padSz; ++i)
117  {
118  for(int j=0; j < imOut.cols(); ++j)
119  {
120  if( j < padSz )
121  {
122  imOut(i,j) = imIn(0, 0);
123  imOut( imOut.rows()-1-i, j) = imIn( dim1-1, 0);
124  }
125  else if( j >= dim2 + padSz)
126  {
127  imOut(i,j) = imIn(0, dim2-1);
128  imOut( imOut.rows()-1-i, j) = imIn( dim1-1, dim2-1);
129  }
130  else
131  {
132  imOut( i, j ) = imOut( padSz, j );
133  imOut( imOut.rows() - padSz + i, j ) = imOut( padSz + dim1-1, j );
134  }
135  }
136  }
137 
138  //Now do the top and bottom rows.
139  for(int i=padSz; i < dim1+padSz; ++i)
140  {
141  for(int j=0; j< padSz; ++j)
142  {
143  imOut(i,j) = imOut(i, padSz);
144  imOut(i, imOut.cols()-1-j) = imOut(i, imOut.cols()-1-padSz);
145  }
146  }
147 
148 
149  return 0;
150 }

290 {
291 
292  slopes.measurement.resize(1, _measurementSize );
293 
294  int k = 0;
295 
296  for(int i=0; i< wfsImage.image.rows(); ++i)
297  {
298  for(int j=0; j < wfsImage.image.cols(); ++j)
299  {
300  if( (*_pupil)(i,j) == 0) continue;
301 
302  slopes.measurement(0, k) = wfsImage.image(i,j);
303  ++k;
304  }

68 realT phiInt_basic (realT phi, void * params)
69 {
70 
71  fourierCovariance<realT, aosysT> * Pp = (fourierCovariance<realT, aosysT> *) params;
72 
73  realT L0 = Pp->aosys->atm.L_0();
74  realT D = Pp->aosys->D();
75 
76  int p = Pp->p;
77  realT m = Pp->m;
78  realT n = Pp->n;
79 
80  int pp = Pp->pp;
81  realT mp = Pp->mp;
82  realT np = Pp->np;
83 
84  realT k = Pp->k;
85 
86  realT cosp = cos(phi);
87  realT sinp = sin(phi);
88 
89  /*** no prime ***/
90  realT kmn_p, kmn_m;
91  realT Ji_mn_p, Ji_mn_m;
92  realT Q_mn;//, Qs_mn;
93 
94  kmn_p = sqrt( pow(k*cosp + m/D, 2) + pow(k*sinp + n/D, 2));
95  kmn_m = sqrt( pow(k*cosp - m/D, 2) + pow(k*sinp - n/D, 2));
96 
97  Ji_mn_p = math::func::jinc(pi<realT>()*D*kmn_p);
98  Ji_mn_m = math::func::jinc(pi<realT>()*D*kmn_m);
99 
100 
101  realT N = 1./sqrt(0.5 + p*math::func::jinc(2*pi<realT>()*sqrt(m*m+n*n)));
102 
103  Q_mn = N*(Ji_mn_p + p*Ji_mn_m);
104 
105  /*** primed ***/
106  realT kmpnp_p, kmpnp_m;
107  realT Ji_mpnp_p, Ji_mpnp_m;
108  realT Q_mpnp;//, Qs_mpnp;
109 
110  kmpnp_p = sqrt( pow(k*cosp + mp/D, 2) + pow(k*sinp + np/D, 2));
111  kmpnp_m = sqrt( pow(k*cosp - mp/D, 2) + pow(k*sinp - np/D, 2));
112 
113 
114  Ji_mpnp_p = math::func::jinc(pi<realT>()*D*kmpnp_p);
115  Ji_mpnp_m = math::func::jinc(pi<realT>()*D*kmpnp_m);
116 
117  realT Np = 1./sqrt(0.5 + pp*math::func::jinc(2*pi<realT>()*sqrt(mp*mp+np*np)));
118 
119  Q_mpnp = Np*(Ji_mpnp_p + pp*Ji_mpnp_m);
120 
121  realT P = Pp->aosys->psd(Pp->aosys->atm, k, 0, 1.0); //vonKarmanPSD(k, D, L0, Pp->subPiston, Pp->subTipTilt);
122 
123  return P*k*(Q_mn*Q_mpnp);
124 }

127  * -# The number of input photons for pixel (i,j) is \f$ n_{ie} = (in(i,j) + darkCurrent)*expTime + cic \f$

314 template<typename floatT>
315 void ccdDetector<floatT>::exposeImage(imageT & out, imageT & in)
316 {
317 
318  using poisson_param_t = typename std::poisson_distribution<int>::param_type;
319  using gamma_param_t = typename std::gamma_distribution<floatT>::param_type;
320 
321 
322  out.resize(_rows, _cols);
323 
324  imageDownSample(out, in);
325 
326  if(_noNoise) return;
327 
328  for(int i=0;i<_rows;++i)
329  {
330  for(int j=0; j<_cols;++j)
331  {
332  floatT charge = (out(i,j)*_qe + _darkCurrent) * _expTime + _cic;
333 
334  if(charge > 1000)
335  {
336  out(i,j) = charge + normVar*sqrt(charge);
337  }
338  else
339  {
340  poissonVar.distribution.param(poisson_param_t{charge});
341  out(i,j) = poissonVar;
342  }
343 
344  if(_gain > 1)
345  {
346  gammaVar.distribution.param(gamma_param_t{out(i,j), _gain});
347 
348 
349  out(i,j) = gammaVar;
350 
351  }
352 
353  out(i,j) += normVar*_ron;
354 
355  if(_gain > 1) out(i,j) /= _gain;
356 
357  out(i,j) = round(out(i,j));
358 
359  }
360  }

26 /// The Ideal Coronagraph
27 /** A simple toy coronagraph which operates only the pupil plane, subtracting the energy-minimizing wavefront \cite cavarroc_2006. For an on-axis source with

29  * for modeling and analysis, and it has been shown that real-world optimized coronagraphs often approach ideal performance \cite sauvage_2010 [and Males in prep].

169 template<typename _realT>
170 int idealCoronagraph<_realT>::propagate( complexFieldT & pupilPlane )
171 {
172  if( pupilPlane.rows() != _realPupil.rows() || pupilPlane.cols() != _realPupil.cols())
173  {
174  mxError("idealCoronagraph", MXE_SIZEERR, "pupilPlane wavefront size does not match realPupil");
175  return -1;
176  }
177 
178  Eigen::Map<Eigen::Array<complexT,-1,-1> > eigWf(pupilPlane.data(), pupilPlane.cols(), pupilPlane.rows());
179  Eigen::Map<Eigen::Array<realT,-1,-1> > eigPup(_realPupil.data(), _realPupil.cols(), _realPupil.rows());
180 
181  eigWf = eigWf - ((eigWf*eigPup).sum()/(eigPup*eigPup).sum())*eigPup;
182 
183  return 0;
184 }

38 /// Get the Zernike coefficients n,m corrresponding the Noll index j.
39 /** Calculates the single index j for (n,m) following Noll (1976) \cite noll_1976
40  * See also: http://en.wikipedia.org/wiki/Zernike_polynomials
41  *
42  * \param[out] n the radial index of the Zernike polynomial
43  * \param[out] m the azimuthal index of the Zernnike polynomial
44  * \param[in] j the Noll index, \f$j > 0\f$
45  *
46  * \retval 0 on success
47  * \retval -1 on error (j < 1)
48  */
49 int noll_nm( int & n, int & m, int j )
50 {
51  if( j < 1)
52  {
53  mxError("noll_nm", MXE_INVALIDARG, "The Noll index j cannot be less than 1 in the Zernike polynomials");
54  return -1;
55  }
56 
57 
58  n = ceil(-1.5 + sqrt(0.25 + 2*j) - 1e-10); // 1e-10 is to avoid wrong rounding due to numerical precision
59 
60  int jrem = j - (n * (n+1)/2+1);
61  m = (jrem + (jrem % 2) * abs( (n % 2)-1) + fabs( (jrem % 2)-1) * (n % 2)) * (-boost::math::sign( (j % 2)-0.5));
62 
63  return 0;
64 }

66 /// Calculate the coefficients of a Zernike radial polynomial
67 /**
68  * \param[out] c is allocated to length \f$ 0.5(n-m)+1\f$ and filled with the coefficients.
69  * \param[in] n is the radial index of the Zernike polynomial.
70  * \param[in] m is the azimuthal index of the Zernike polynomial.
71  *
72  * \retval 0 on success
73  * \retval -1 on error
74  *
75  * \tparam realT is a real floating type
76  */
77 template<typename realT>
78 int zernikeRCoeffs( std::vector<realT> & c, int n, int m)
79 {
80  m = abs(m);
81 
82  if( n < m)
83  {
84  mxError("zernikeRCoeffs", MXE_INVALIDARG, "n cannot be less than m in the Zernike polynomials");
85  return -1;
86  }
87 
88  //If odd, it's 0.
89  if( (n-m) % 2 > 0)
90  {
91  c.resize(1,0);
92  return 0;
93  }
94 
95  int ul = 0.5*(n-m) + 1;
96 
97  c.resize(ul);
98 
99  for(int k=0; k < ul; ++k)
100  {
101  c[k] = pow(-1.0, k) * boost::math::factorial<realT>(n - k) / ( boost::math::factorial<realT>(k) * boost::math::factorial<realT>(0.5*(n+m) - k)* boost::math::factorial<realT>(0.5*(n-m) - k));
102  }
103 
104  return 0;
105 }

120 realT zernikeR( realT rho, int n, int m, std::vector<realT> & c )
121 {
122  m = abs(m);
123 
124  //If odd, it's 0.
125  if( (n-m) % 2 > 0)
126  {
127  c.resize(1,0);
128  return 0.0;
129  }

262 ///Fill in an Eigen-like array with a Zernike polynomial
263 /** Sets any pixel which is at rad <= r < rad+0.5 pixels to rho = 1, to be consistent with mx::circularPupil
264  *
265  * \param[out] arr is the allocated array with an Eigen-like interface. The rows() and cols() members are used to size the polynomial.
266  * \param[in] n is the radial index of the polynomial
267  * \param[in] m is the azimuthal index of the polynomial
268  * \param[in] xcen is the x coordinate of the desired center of the polynomial, in pixels
269  * \param[in] ycen is the y coordinate of the desired center of the polynomial, in pixels
270  * \param[in] rad [optional] is the desired radius. If rad <= 0, then the maximum radius based on dimensions of m is used.
271  */
272 template<typename arrayT, int overscan=2>
273 int zernike( arrayT & arr,
274  int n,
275  int m,
276  typename arrayT::Scalar xcen,
277  typename arrayT::Scalar ycen,
278  typename arrayT::Scalar rad = -1 )
279 {
280  typename arrayT::Scalar x;
281  typename arrayT::Scalar y;
282  typename arrayT::Scalar r, rho;
283  typename arrayT::Scalar phi;
284 
285  std::vector<typename arrayT::Scalar> c;
286 
287  if(zernikeRCoeffs(c, n, m) < 0) return -1;
288 
289  size_t l0 = arr.rows();
290  size_t l1 = arr.cols();
291 
292  if(rad <= 0) rad = 0.5*std::min(l0-1, l1-1);
293 
294  for(size_t i=0; i < l0; ++i)
295  {
296  for(size_t j=0; j < l1; ++j)
297  {
298  x = i - xcen;
299  y = j - ycen;
300 
301 
302  r = std::sqrt( x*x + y*y );
303 
304  //This is to be consistent with mx::circularPupil while still respecting the Zernike rules
305  if(r > rad && r <= rad+(1.0/overscan)) r = rad;
306 
307  rho = r / rad;
308 
309  if(rho <= 1.0)
310  {
311  phi = std::atan2( y, x);
312  arr(i,j) = zernike(rho, phi, n, m, c);
313  }
314  else
315  {
316  arr(i,j) = 0.0;
317  }
318  }
319  }
320  return 0;
321 }

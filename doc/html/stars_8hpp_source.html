25 ///Parse a main sequence spectral type string into a numeric code.
26 /** Expects standard spectral type strings such as "O5V" or "F2.5" or "G8.5V".
27  * The numeric code starts at 0 for "O0V", 10 for "A0V", through 90 for "Y0". The subtypes are added to this value. For instance,
28  * "G8.5V" becomes 48.5.
29  *
30  * Only works for main sequence types. Any other spectral type, such as MK class I-IV, will return -1.
31  *
32  */
33 template<typename realT=float>
34 realT numSpType( std::string spType /**< [in] The spectral type string to parse.*/)
35 {
36 
37  spType = removeWhiteSpace(spType);
38 
39  spType = toUpper(spType);
40 
41 
42  realT num = 0;
43 
44  switch( spType[0] )
45  {
46  case 'O':
47  num = 0;
48  break;
49  case 'B':
50  num = 10;
51  break;
52  case 'A':
53  num = 20;
54  break;
55  case 'F':
56  num = 30;
57  break;
58  case 'G':
59  num = 40;
60  break;
61  case 'K':
62  num = 50;
63  break;
64  case 'M':
65  num = 60;
66  break;
67  case 'L':
68  num = 70;
69  break;
70  case 'T':
71  num = 80;
72  break;
73  case 'Y':
74  num = 90;
75  break;
76  default:
77  return -1;
78  }
79 
80  if(spType.size() < 2) return -1;
81  if(!isdigit(spType[1])) return -1;
82 
83  int i=1;
84  while( i < spType.size() && ( isdigit(spType[i]) || spType[i] == '.')) ++i;
85 
86  std::string subType = spType.substr(1, i-1);
87 
88  num += convertFromString<realT>(subType);
89 
90  if(spType.size() == i) return num;
91  if(spType[i] == 'V') return num;
92 
93  return -1;
94 }

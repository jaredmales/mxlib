74  ///Load the spectrum and interpolate it on a wavelength scale.
75  template<typename gridT>
76  int setSpectrum( gridT & lambda )
77  {
78  std::vector<realT> rawLambda;
79  std::vector<realT> rawSpectrum;
80 
81  if(_dataDir == "") _dataDir = ".";
82 
83  std::string path = _dataDir + "/" + spectrumT::fileName(_name);
84 
85  spectrumT::readSpectrum(rawLambda, rawSpectrum, path);
86 
87  for(int i=0; i < rawLambda.size(); ++i)
88  {
89  rawLambda[i] /= spectrumT::wavelengthUnits/(units::length);
90  rawSpectrum[i] /= spectrumT::fluxUnits/(units::energy/(units::time * units::length * units::length * units::length));
91  }
92 
93  _spectrum.clear();
94 
95  mx::gsl_interpolate(gsl_interp_linear, rawLambda, rawSpectrum, lambda, _spectrum);
96 
97  for(int i=0; i < lambda.size(); ++i)
98  {
99  if( !std::isnormal(_spectrum[i])) _spectrum[i] = 0;
100  }
101 
102  return 0;
103  }

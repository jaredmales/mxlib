301 void appConfigurator::parseCommandLine( int argc,
302  char ** argv,
303  const std::string & oneTarget
304  )
305 {
306  if(argc == 0) return;
307 
308  clOptions clOpts;
309 
310  //First load the options into the clOptions parser
311  targetIterator it;
312  for(it = targets.begin(); it != targets.end(); ++it)
313  {
314  if(it->second.shortOpt == "" && it->second.longOpt == "") continue; //No command line opts specified.
315 
316  if(oneTarget != "" && it->second.name != oneTarget) continue;
317 
318  clOpts.add(it->second.name,it->second.shortOpt.c_str(),it->second.longOpt.c_str(), it->second.clType);
319  }
320 
321  //Then load the command-line-only options.
322  clOnlyTargetIterator cloit;
323  for(cloit = clOnlyTargets.begin(); cloit != clOnlyTargets.end(); ++cloit)
324  {
325  if(cloit->shortOpt == "" && cloit->longOpt == "") continue; //Nothing to add?
326 
327  if(oneTarget != "" && cloit->name != oneTarget) continue;
328 
329  clOpts.add(cloit->name,cloit->shortOpt.c_str(),cloit->longOpt.c_str(), cloit->clType);
330  }
331 
332  //If we get here and nothing to do, get out
333  if(clOpts.nOpts == 0)
334  {
335  return;
336  }
337 
338  //Now we parse
339  clOpts.parse(argc, argv, &nonOptions);
340 
341  //And then load the results in the config target map.
342  for(it = targets.begin(); it != targets.end(); ++it)
343  {
344  if(clOpts.optSet(it->second.name))
345  {
346  std::vector<std::string> args;
347 
348  clOpts.getAll(args, it->second.name);
349  it->second.values.insert( it->second.values.end(), args.begin(), args.end());
350  it->second.set = true;
351  }
352  }

418 int appConfigurator::get( std::vector<typeT> & v,
419  const std::string & name,
420  int i
421  )
422 {
423 
424  if(!isSet(name)) return 0;
425 
426  if( targets[name].values.size() <= i) return -1;
427 
428  std::string s;
429  if( get<std::string>(s, name, i) < 0) return -1;
430 
431  //Case that s was set to be empty.
432  if(s.size() == 0)
433  {
434  v.clear();
435  return 0;
436  }
437 
438  int st;
439  int com;
440 
441  st = 0;
442  com = s.find(',', st);
443 
444  v.clear();
445 
446  while(com != std::string::npos)
447  {
448  v.push_back( convertFromString<typeT>(s.substr(st, com-st)) );
449  st = com + 1;
450  com = s.find(',', st);
451  }
452  v.push_back( convertFromString<typeT>(s.substr(st, s.size()-st)));
453 
454  return 0;

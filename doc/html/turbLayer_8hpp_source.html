187 template<typename realT>
188 void turbLayer<realT>::shift( realT dt )
189 {
190  int wdx, wdy;
191  realT ddx, ddy;
192 
193  shiftPhase.setZero();
194 
195  for(int i=0; i < _nCombo; ++i)
196  {
197  ddx = _x0[i] + _dx*dt;
198  ddy = _y0[i] + _dy*dt;
199 
200  wdx = (int) trunc(ddx);
201  ddx -= wdx;
202  wdy = (int) trunc(ddy);
203  ddy -= wdy;
204 
205  wdx %= _scrnSz;
206  wdy %= _scrnSz;
207 
208  //Check for a new whole-pixel shift
209  if(wdx != _last_wdx[i] || wdy != _last_wdy[i])
210  {
211  //Need a whole pixel shift
212  improc::imageShiftWP(shiftPhaseWP[i], phase, wdx, wdy);
213  }
214 
215  //Do the sub-pixel shift
216  improc::imageShift( shiftPhaseWork, shiftPhaseWP[i], ddx, ddy, improc::cubicConvolTransform<realT>(-0.5));
217  shiftPhase += shiftPhaseWork;
218 
219  _last_wdx[i] = wdx;
220  _last_wdy[i] = wdy;
221  }
222  shiftPhase /= sqrt(_nCombo);
223 }

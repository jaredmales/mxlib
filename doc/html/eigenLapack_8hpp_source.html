307 ///Compute the SVD of an Eigen::Array using LAPACK's xgesdd
308 /** Computes the SVD of A, \f$ A = U S V^T \f$.
309  *
310  * \param[out] U the A.rows() x A.rows() left matrix
311  * \param[out] S the A.cols() x 1 matrix of singular values
312  * \param[out] VT the A.cols() x A.cols() right matrix, note this is the transpose.
313  * \param[in] A the input matrix to be decomposed
314  *
315  * \returns
316  * \parblock
317  * 0 on success
318  * -i on error in ith parameter (from LAPACK xgesdd)
319  * >0 did not converge (from LAPACK xgesdd)
320  * \endparblock
321  *
322  * \tparam dataT is either float or double.
323  *
324  * \ingroup eigen_lapack
325  */
326 template<typename dataT>
327 MXLAPACK_INT eigenGESDD( Eigen::Array<dataT,-1,-1> & U, Eigen::Array<dataT,-1,-1> & S, Eigen::Array<dataT,-1,-1> & VT, Eigen::Array<dataT,-1,-1> & A )
328 {
329  char JOBZ = 'A';
330  MXLAPACK_INT M = A.rows();
331  MXLAPACK_INT N = A.cols();
332  MXLAPACK_INT LDA = M;
333  S.resize(N,1);
334  U.resize(M,M);
335  MXLAPACK_INT LDU = M;
336  VT.resize(N,N);
337  MXLAPACK_INT LDVT = N;
338 
339  dataT wkOpt;
340  MXLAPACK_INT LWORK = -1;
341 
342  MXLAPACK_INT * IWORK = new MXLAPACK_INT[8*M];
343  MXLAPACK_INT INFO;
344 
345  math::gesdd<dataT>(JOBZ, M, N, A.data(), LDA, S.data(), U.data(), LDU, VT.data(), LDVT, &wkOpt, LWORK, IWORK, INFO);
346 
347  LWORK = wkOpt;
348  //delete WORK;
349  dataT *WORK = new dataT[LWORK];
350 
351  INFO = math::gesdd<dataT>(JOBZ, M, N, A.data(), LDA, S.data(), U.data(), LDU, VT.data(), LDVT, WORK, LWORK, IWORK, INFO);
352 
353  delete WORK;
354  delete IWORK;
355 
356  return INFO;

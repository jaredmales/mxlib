134 /// Convert a transmission profile to relative spectral response.
135 /** The relative spectral response is given by
136  *
137  * \f$ T_{RSR}(\lambda) = T(\lambda)\lambda / \mbox{max}[T(\lambda)\lambda] \f$
138  *
139  * where \f$ T(\lambda) \f$ is the energy transmission profile.
140  * This is appropriate for photon detectors (like CCDs). See Bessell, PASP, 112, 961 (2000).
141  *
142  * \param lambda [in] The wavelength scale
143  * \param trans [in] The transmission profile
144  *
145  * \tparam dataT the type of the data
146  */
147 template<typename dataT>
148 void astrofiltRSR( std::vector<dataT> & lambda,
149  std::vector<dataT> & trans )
150 {
151  dataT max = 0;
152 
153  for(int i=0;i<lambda.size(); ++i)
154  {
155  trans[i] = trans[i]*lambda[i];
156  if(trans[i] > max) max = trans[i];
157  }
158 
159  for(int i=0;i<lambda.size(); ++i)
160  {
161  trans[i] /= max;
162  }
163

211 /// Characterize the wavelength of a filter transmission curve.
212 /** For a transmission curve given by \f$ T(\lambda ) \f$ The central wavelength is defined as
213  * \f$ \lambda_0 = \frac{1}{w_{eff}}\int T(\lambda ) \lambda d\lambda \f$
214  * where the effective width is defined by
215  * \f$ w_{eff} = \int T(\lambda ) d\lambda \f$
216  *
217  * The full-width at half-maximum, FWHM, is the distance between the points at 50% of maximum \f$ T(\lambda) \f$.
218  *
219  * \param lambda0 [out] the central wavelength of the filter
220  * \param fwhm [out] the full-width at half-maximum of the filter profile
221  * \param weff [out] the effective width of the filter
222  * \param lambda [in] the wavelength grid of the profile
223  * \param trans [in] the transmission of the profile at each wavelength
224  *
225  * \tparam dataT the data type in which calculations are performed.
226  */
227 template<typename dataT>
228 void astrofiltCharTrans( dataT & lambda0,
229  dataT & fwhm,
230  dataT & weff,
231  std::vector<dataT> & lambda,
232  std::vector<dataT> & trans )
233 {
234 
235  weff = 0;
236 
237  for(int i=0; i< lambda.size()-1; ++i) weff += trans[i]*(lambda[i+1]-lambda[i]);
238  weff += trans[trans.size()-1]* (lambda[lambda.size()-1] -lambda[lambda.size()-2]);
239 
240  lambda0 = 0;
241  for(int i=0; i< lambda.size()-1; ++i) lambda0 += lambda[i]*trans[i]*(lambda[i+1]-lambda[i]);
242  lambda0 += lambda[trans.size()-1]* trans[trans.size()-1] * (lambda[lambda.size()-1] -lambda[lambda.size()-2]);
243 
244  lambda0 /= weff;
245 
246 
247  dataT max, left, right;
248 
249  max = 0;
250  for(int i=0; i< lambda.size(); ++i)
251  {
252  if(trans[i] > max) max = trans[i];
253  }
254 
255  int i=1;
256  while(trans[i] < 0.5*max && i < lambda.size()) ++i;
257  left = lambda[i-1] + (0.5 - trans[i-1])* ( (lambda[i] - lambda[i-1]) / (trans[i]-trans[i-1]));
258 
259  while(trans[i] < max && i < lambda.size()) ++i;
260 
261  while(trans[i] > 0.5*max && i < lambda.size()) ++i;
262  right = lambda[i-1] + (0.5 - trans[i-1])* ( (lambda[i] - lambda[i-1]) / (trans[i]-trans[i-1]));

67 ///Calculate the separation and true anomaly for an orbit at the specified mean anomaly.
68 /**
69  * \tparam realT is the type used for arithmetic
70  *
71  * \returns the numer of iterations in solving Kepler's equation, if successful.
72  * \returns -1 if itmax is exceeded.
73  */
74 template<typename realT>
75 long orbitElements( realT & r, ///< [out] is the orbital separation
76  realT & f, ///< [out] is the true anomaly
77  realT E, ///< [out] is the eccentric anomaly, provided since it is free
78  realT D, ///< [out] is the error after the last iteration of the Kepler solution.
79  realT e, ///< [in] is the eccentricity
80  realT M, ///< [in] is the mean anomaly
81  realT a, ///< [in] is the semi-major axis
82  realT tol=KEPLER_TOL, ///< [in] is the desired tolerance
83  long itmax=KEPLER_ITMAX ///< [in] is the maximum number of iterations
84  )
85 {
86  long its;
87 
88  if(e < 0.0)
89  {
90  std::cerr << "e < 0 in rf_elements\n";
91  return -1;
92  }
93 
94  if(e == 1.0)
95  {
96  std::cerr << "e = 1 in rf_elements, which is not currently handled.\n";
97  return -1;
98  }
99 
100  its = solve_kepler(E, D, e, M, tol, itmax);
101 
102  if(e > 1.0)
103  {
104  f = 2.0*std::atan(std::sqrt((e+1.0)/(e-1.0))*std::tanh(E/2.0));
105  r = a*(1.0-e*std::cosh(E));
106  }
107  else if(e<1.0)
108  {
109  if(e == 0.0)
110  {
111  f = M;
112  r = a;
113  }
114  else
115  {
116  f = 2.0*std::atan(std::sqrt((1.0+e)/(1.0-e))*std::tan(E/2.0));
117  r = a * (1.0-e*std::cos(E));
118  }
119  }
120 
121  return its;
122 }

155 ///Calculate various quantities of an orbit given the keplerian elements and a vector of times.
156 /** Only those quantities with non-null pointers are actually calculated.
157  *
158  * The orbital parameters with dimensions, (a, P, and t0) must be in consistent units.
159  *
160  * \tparam vectorT a type whose elements are accessed with the [] operator. No other requirements are placed on this type, so it could be a native pointer (i.e. double *).
161  * \tparam realT the type in which to perform calculations. Does not need to match the storage type of vectorT.
162  *
163  * \retval -1 on error calculating the orbit (from rf_elements)
164  * \retval 0 on success.
165  */
166 template<typename vectorT, typename realT>
167 int orbitCartesianWork( vectorT * x, ///< [out] [optional] If not NULL, will be the projected x positions of the orbit. Must be at least as long as t.
168  vectorT * y, ///< [out] [optional] If not NULL, will be the projected y positions of the orbit. Must be at least as long as t.
169  vectorT * z, ///< [out] [optional] If not NULL, will be the projected z positions of the orbit. Must be at least as long as t.
170  vectorT * r, ///< [out] [optional] If not NULL, will be the separation of the orbit. Must be at least as long as t.
171  vectorT * rProj, ///< [out] [optional] If not NULL, will be the projected separation of the orbit. Must be at least as long as t.
172  vectorT * f, ///< [out] [optional] If not NULL, will be the true anomaly of the orbit. Must be at least as long as t.
173  vectorT * cos_alf, ///< [out] [optional] If not NULL, will be the phase angle of the orbit. Must be at least as long as t.
174  vectorT * phi, ///< [out] [optional] If not NULL, will be the Lambertian phase function. Must be at least as long as t.
175  const vectorT & t, ///< [in] the times at which to calculate the projected positions
176  const size_t N, ///< [in] the number of points contained in t, and allocated in nx, ny, nz. Avoids requiring a size() member (i.e. if vectorT is native pointers).
177  realT a, ///< [in] the semi-major axis of the orbit.
178  realT P, ///< [in] the orbital period.
179  realT e, ///< [in] the eccentricity of the orbit.
180  realT t0, ///< [in] the time of pericenter passage of the orbit.
181  realT i, ///< [in] the inclination of the orbit [radians].
182  realT w, ///< [in] the argument of pericenter of the orbit [radians].
183  realT W ///< [in] the longitude of the ascending node of the orbit [radians].
184  )
185 {
186  long rv;
187  realT _M, _r, _f, _E, _D;
188 
189  //Just do these calcs once
190  realT cos_i = cos(i);
191  realT sin_i = sin(i);
192  realT cos_W = cos(W);
193  realT sin_W = sin(W);
194 
195  realT cos_wf, sin_wf;
196 
197  for(size_t j=0; j < N; j++)
198  {
199  _M = orbitMeanAnol(t[j], t0, P);
200 
201  rv = orbitElements(_r, _f, _E, _D, e, _M, a);
202 
203  if(rv < 0) return -1;
204 
205  //one calc
206  cos_wf = cos(w+_f);
207  sin_wf = sin(w+_f);
208 
209  //Assumption is that these will be optimized away if NULL
210  if(x) (*x)[j] = _r*(cos_W*cos_wf-sin_W*sin_wf*cos_i);
211  if(y) (*y)[j] = _r*(sin_W*cos_wf+cos_W*sin_wf*cos_i);
212  if(z) (*z)[j] = _r*sin_wf*sin_i;
213  if(r) (*r)[j] = _r;
214  if(rProj)
215  {
216  if(x && y)
217  {
218  (*rProj)[j] = sqrt((*x)[j]*(*x)[j] + (*y)[j]*(*y)[j]);
219  }
220  else if(x && !y)
221  {
222  realT _y = _r*(sin_W*cos_wf+cos_W*sin_wf*cos_i);
223  (*rProj)[j] = sqrt((*x)[j]*(*x)[j] + _y*_y);
224  }
225  else if(!x && y)
226  {
227  realT _x = _r*(cos_W*cos_wf-sin_W*sin_wf*cos_i);
228  (*rProj)[j] = sqrt( _x*_x + (*y)[j]*(*y)[j]);
229  }
230  else
231  {
232  realT _x = _r*(cos_W*cos_wf-sin_W*sin_wf*cos_i);
233  realT _y = _r*(sin_W*cos_wf+cos_W*sin_wf*cos_i);
234  (*rProj)[j] = sqrt( _x*_x + _y*_y);
235  }
236  }
237 
238  if(f) (*f)[j] = _f;
239 
240  if(cos_alf) (*cos_alf)[j] = sin_wf*sin_i;
241 
242  if(phi) (*phi)[j] = orbitLambertPhase<realT>(sin_wf*sin_i);
243 
244  }
245  return 0;
246 }

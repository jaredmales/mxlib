591 template<typename _realT>
592 int HCIobservation<_realT>::readFiles()
593 {
594  if(fileList.size() == 0)
595  {
596  mxError("HCIobservation", MXE_FILENOTFOUND, "The fileList has 0 length, there are no files to be read.");
597  return -1;
598  }
599 
600  //First make the list deletions
601  if(!filesDeleted)
602  {
603  if(deleteFront > 0)
604  {
605  fileList.erase(fileList.begin(), fileList.begin()+deleteFront);
606  }
607 
608  if(deleteBack > 0)
609  {
610  fileList.erase(fileList.end()-deleteBack, fileList.end());
611  }
612  filesDeleted = true;
613  }
614 
615  if( fileList.size() == 0)
616  {
617  mxError("HCIobservation", MXE_FILENOTFOUND, "The fileList has 0 length, there are no files to be read after deletions.");
618  return -1;
619  }
620 
621  if(qualityFile != "")
622  {
623  if( threshold() < 0) return -1;
624 
625  if(thresholdOnly)
626  {
627  std::cout << "#Files which passed thresholding:\n";
628  for(int i=0; i<fileList.size(); ++i)
629  {
630  std::cout << fileList[i] << "\n";
631  }
632 
633  exit(0);
634  }
635  }
636 
637  if( fileList.size() == 0)
638  {
639  mxError("HCIobservation", MXE_FILENOTFOUND, "The fileList has 0 length, there are no files to be read after thresholding.");
640  return -1;
641 
642  }
643 
644  if(weightFile != "")
645  {
646  if(readWeights() < 0) return -1;
647  }
648 
649 
650 
651  Eigen::Array<realT, Eigen::Dynamic, Eigen::Dynamic> im;
652 
653  fitsFile<realT> f(fileList[0]);
654 
655  f.read(im);
656 
657  fitsHeader head;
658 
659  if(MJDKeyword != "") head.append(MJDKeyword);
660 
661  for(int i=0;i<keywords.size();++i)
662  {
663  head.append(keywords[i]);
664  }
665 
666  /*----- Append the HCI keywords to propagate them if needed -----*/
667 
668  heads.clear(); //This is necessary to make sure heads.resize() copies head on a 2nd call
669  heads.resize(fileList.size(), head);
670 
671  if(imSize > 0)
672  {
673  //the +0.1 is just to make sure we don't have a problem with precision (we shouldn't)/
674  f.setReadSize( floor(0.5*(im.rows()-1) - 0.5*(imSize-1) +0.1), floor(0.5*(im.cols()-1.0) - 0.5*(imSize-1.0)+0.1), imSize, imSize);
675  im.resize(imSize, imSize);
676  }
677 
678  imc.resize(im.rows(), im.cols(), fileList.size());
679 
680  t_load_begin = get_curr_time();
681 
682  /** \bug It is this step that is really slow during file reads *sometimes*
683  */
684  f.read(imc.data(), heads, fileList);
685 
686  f.setReadSize();
687 
688  if(MJDKeyword != "")
689  {
690  imageMJD.resize(heads.size());
691 
692  if(MJDisISO8601)
693  {
694  for(int i=0;i<imageMJD.size();++i)
695  {
696  imageMJD[i] = mx::ISO8601date2mjd(heads[i][MJDKeyword].String());
697  }
698  }
699  else
700  {
701  for(int i=0;i<imageMJD.size();++i)
702  {
703  imageMJD[i] = heads[i][MJDKeyword].Value<realT>()*MJDUnits;
704  }
705  }
706  }
707 
708  t_load_end = get_curr_time();
709 
710  Nims = imc.planes();
711  Nrows = imc.rows();
712  Ncols = imc.cols();
713  Npix = imc.rows()*imc.cols();
714 
715  /*** Now do the post-read actions ***/
716  if( postReadFiles() < 0) return -1;
717 
718  /*** Load the mask ***/
719  if( maskFile != "")
720  {
721  fitsFile<realT> ff;
722  ff.read(mask, maskFile);
723 
724  makeMaskCube();
725  }
726 
727  if(!skipPreProcess)
728  {
729  /*** Now do any pre-processing ***/
730  if(preProcess_beforeCoadd) preProcess();
731 
732  if(coaddCombineMethod != HCI::noCombine)
733  {
734  coaddImages();
735  }
736 
737  /*** Now do any pre-processing if not done already***/
738  if(!preProcess_beforeCoadd) preProcess();
739 
740  outputPreProcessed();
741  }
742  filesRead = true;
743 }

745 template<typename _realT>
746 int HCIobservation<_realT>::threshold()
747 {
748  std::cerr << "Thresholding . . .\n";
749  if(qualityFile == "")
750  {
751  mxError("HCIobservation::threshold", MXE_PARAMNOTSET, "qualityFile not set");
752  return -1;
753  }
754 
755  int origsize = fileList.size();
756 
757  std::vector<std::string> qfileNames;
758  std::vector<realT> imQ;
759 
760  //Read the quality file and load it into a map
761  readColumns(qualityFile, qfileNames, imQ);
762 
763  std::map<std::string, realT> quality;
764  for(int i=0;i<qfileNames.size();++i) quality[basename(qfileNames[i].c_str())] = imQ[i];
765 
766  realT q;
767 
768  for(int i=0; i<fileList.size(); ++i)
769  {
770  try

860 template<typename _realT>
861 void HCIobservation<_realT>::coaddImages()
862 {
863  //Validate setup
864  if(coaddMaxImno <=0 && coaddMaxTime <= 0) return;
865  //Validate combine method
866  if(coaddCombineMethod == HCI::noCombine) return;
867 
868  t_coadd_begin = get_curr_time();
869 
870  std::cerr << "coadding raw images\n";
871 
872  std::vector<eigenImageT> coadds;
873 
874  //We do all math here in double, to avoid losing precision
875  std::vector<double> avgMJD;
876  std::vector<std::vector<double> > avgVals;
877 
878  int combineMethod = HCI::medianCombine;
879  if(coaddCombineMethod == HCI::meanCombine) combineMethod = HCI::meanCombine;
880 
881  //Index range of images for next coadd
882  int im0, imF;
883  im0 = 0;
884  imF = 1;
885 
886  //Accumulate images to coadd into a cube
887  eigenCube<realT> imsToCoadd;
888 
889  //Temporary for combination.
890  eigenImageT coadd;
891 
892  //Accumulate values
893  double initMJD;
894  std::vector<double> initVals;
895  initVals.resize(coaddKeywords.size());
896 
897  while(im0 < Nims)
898  {
899  //Initialize the accumulators
900  initMJD = imageMJD[im0];
901 
902  for(int i=0;i<coaddKeywords.size(); ++i)
903  {
904  initVals[i] = heads[im0][coaddKeywords[i]].Value<double>();
905  }
906 
907  //Now increment imF, then test whether each variable is now outside the range
908  bool increment = true;
909  while(increment == true)
910  {
911  ++imF;
912 
913  if(imF >= Nims)
914  {
915  imF = Nims;
916  increment = false;
917  break;
918  }
919 
920  if(imF-im0 > coaddMaxImno && coaddMaxImno > 0)
921  {
922  --imF;
923  increment = false;
924  break;
925  }
926 
927  ///\todo this should really include end of exposure too.
928  if( (imageMJD[imF] - imageMJD[im0])*86400.0 > coaddMaxTime && coaddMaxTime > 0)
929  {
930  --imF;
931  increment = false;
932  break;
933  }
934 
935  }//while(increment == true)
936  //At this point, imF is the first image NOT to include in the next coadd.
937 
938  //Now actually do the accumulation
939  ///\todo this needs to handle averaging of angles
940  for(int imno = im0+1; imno < imF; ++imno)
941  {
942  initMJD += imageMJD[imno];
943  for(int i=0;i<coaddKeywords.size(); ++i)
944  {
945  initVals[i] += heads[imno][coaddKeywords[i]].Value<double>();
946  }
947  }
948 
949  //And then turn them into an average
950  initMJD /= (imF - im0);
951  for(int i=0;i<coaddKeywords.size(); ++i)
952  {
953  initVals[i] /= (imF-im0);
954  }
955 
956  //Extract the images into the temporary
957  imsToCoadd.resize(Nrows, Ncols, imF-im0);
958  for(int i =0; i < (imF-im0); ++i)
959  {
960  imsToCoadd.image(i) = imc.image(im0 + i);
961  }
962 
963  //Here do the combine and insert into the vector
964  if(combineMethod == HCI::medianCombine)
965  {
966  imsToCoadd.median(coadd);
967  }
968  if(combineMethod == HCI::meanCombine)
969  {
970  imsToCoadd.mean(coadd);
971  }
972  coadds.push_back(coadd);
973 
974  //Insert the new averages
975  avgMJD.push_back(initMJD);
976  avgVals.push_back(initVals);
977 
978  im0 = imF;
979  imF = im0 + 1;
980  }//while(im0 < Nims)
981 
982  //Now resize imc and copy the coadds to the new cube
983  imc.resize(Nrows, Ncols, coadds.size());
984  Nims = coadds.size();
985 
986  for(int i=0;i<Nims;++i) imc.image(i) = coadds[i];
987 
988  Nims = imc.planes();
989  Nrows = imc.rows();
990  Ncols = imc.cols();
991  Npix = imc.rows()*imc.cols();
992 
993  //Now deal with imageMJD and headers
994  imageMJD.erase(imageMJD.begin()+Nims, imageMJD.end());
995  heads.erase(heads.begin()+Nims, heads.end());
996  for(int i=0;i<Nims;++i)
997  {
998  imageMJD[i] = avgMJD[i];
999  for(int j=0;j<coaddKeywords.size(); ++j)
1000  {
1001  heads[i][coaddKeywords[j]].setValue(avgVals[i][j]);
1002  }
1003  }
1004 
1005  t_coadd_end = get_curr_time();
1006 
1007 }//void HCIobservation<_realT>::coaddImages()

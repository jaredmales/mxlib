447  _floatT pi = boost::math::constants::pi<_floatT>();

449  _floatT dang = 2*pi/_modSteps;
450  _floatT dx, dy;
451 
452  tilts.resize(_modSteps);
453 
454  std::cout << "WF Size: " << _wfSz << "\n";
455  std::cout << "WF PS: " << _wfPS << "\n";

460

462  {
463  dx = _modRadius * (_lambda/_D) / mx::fftPlateScale<floatT>(_wfSz, _wfPS, _lambda) * cos(0.5*dang+dang * i);

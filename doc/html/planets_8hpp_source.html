26 ///An ad-hoc planetary mass-to-radius relationship.
27 /** The goal of this function is to provide a radius given an exoplanet mass, for lightly-irradiated exoplanets. By lightly-irradiated we mean (roughly) planet's at
28  * Mercury's separation or further, scaled for stellar luminosity.
29  * Here we make use of the transition from rocky to gas-dominated composition at \f$ 1.6 R_e \f$ identified by Rogers \cite rogers_2015
30  * (see also Marcy et al. (2014) \cite marcy_2014). Below this radius we assume Earth composition and so
31  * \f$ R \propto M^{1/3}\f$. Above this we scale with a power law matched to the mean radius and mass of Uranus and Neptune, which defines the radius between
32  * \f$ 1.6^3 M_\oplus \f$ and this Uranus-Neptune mean point. Above this point we use
33  * a polynomial fit (in log(M)) to points including the Uranus/Neptune mean, Saturn, Jupiter, and above Jupiter's mass the average points from the 4.5 Gyr 1 AU models from
34  * Fortney et al. (2007) \cite fortney_2007. Above 3591.1 \f$ M_\oplus \f$ (\f$\sim 11 M_{jup}\f$) we scale as \f$ M^{-1/8} \f$ based on the curve shown in Fortney et al. (2011) \cite fortney_2010.
35  *
36  * \f$
37  \frac{R}{R_\oplus} =
38  \begin{cases}
39  \left(\frac{M}{M_\oplus}\right)^{1/3}, & M < 4.1 M_\oplus \\
40  0.62\left(\frac{M}{M_\oplus}\right)^{0.67}, & 4.1 M_\oplus \le M < 15.84 M_\oplus \\
41  14.0211 - 44.8414 \log_{10}\left(\frac{M}{M_\oplus}\right) + 53.6554 \log_{10}^2\left(\frac{M}{M_\oplus}\right)
42  -25.3289\log_{10}^3\left(\frac{M}{M_\oplus}\right) + 5.4920\log_{10}^4\left(\frac{M}{M_\oplus}\right) - 0.4586 \log_{10}^5\left(\frac{M}{M_\oplus}\right), & 15.84 \le M < 3591.1 M_\oplus \\
43  32.03 \left(\frac{M}{M_\oplus}\right)^{-1/8}, & 3591.1 M_\oplus \le M
44  \end{cases}
45  \f$
46  *
47  * \image html planet_mrdiag_2017.06.19.png "The ad-hoc mass-to-radius relationship compared to known planets. Blue circles are the Solar system. Black circles indicate expoplanets with Teq below that of a blackbody at Mercury's separation (413 K)."
48  *
49  * This function makes use of the units type system (\ref astrounits) so it can be used with Earth masses, Jupiter masses, kg (SI units), etc.
50  *
51  * \returns the estimated radius of the planet.
52  *
53  * \tparam units is the units-type specifying the units of mass. See \ref astrounits.
54  */
55 template<typename units>
56 typename units::realT planetMass2Radius( typename units::realT mass /**< The mass of the planet. */)
57 {
58  typedef typename units::realT realT;
59 
60  using namespace mx::astro::constants;
61 
62  if( mass < 4.1*massEarth<units>())
63  {
64  return pow( mass/massEarth<units>(), boost::math::constants::third<realT>())*radEarth<units>();
65  }
66  else if( mass < static_cast<realT>(15.84)*massEarth<units>() )
67  {
68  return 0.62*pow( mass/massEarth<units>(), static_cast<realT>(0.67))*radEarth<units>();
69  }
70  else if( mass < 3591.1*massEarth<units>())
71  {
72  realT logM = log10(mass/massEarth<units>());
73 
74  return (static_cast<realT>(14.0211)-static_cast<realT>(44.8414)*logM+static_cast<realT>(53.6554)*pow(logM,2) -
75  static_cast<realT>(25.3289)*pow(logM,3) + static_cast<realT>(5.4920)*pow(logM,4)
76  - 0.4586*pow(logM,5))*radEarth<units>();
77  }
78  else
79  {
80  return static_cast<realT>(32.03)*pow(mass/massEarth<units>(), static_cast<realT>(-0.125))* radEarth<units>();
81  }
82 
83 
84 }

86 ///The planetary mass-to-radius of Fabrycky et al. (2014)..
87 /** A broken power law for low mass planets from Fabrycky et al. (2014)\cite fabrycky_2014 (see also \cite lissauer_2011}.
88  *
89  * \f$
90  \frac{R}{R_e} =
91  \begin{cases}
92  \left(\frac{M}{M_e}\right)^{1/3}, & M < 1 M_e \\
93  \left(\frac{M}{M_e}\right)^{1/2.06}, & 1 M_e \le M
94  \end{cases}
95  \f$
96  *
97  * This makes use of the units type system (\ref astrounits) so it can be used with Earth masses, Jupiter masses, kg (SI units), etc.
98  *
99  * \returns the estimated radius of the planet.
100  *
101  * \tparam units is the units-type specifying the units of mass. See \ref astrounits.
102  */
103 template<typename units>
104 typename units::realT planetMass2RadiusFab2014( typename units::realT mass /**< The mass of the planet. */)
105 {
106  typedef typename units::realT realT;
107 
108  using namespace mx::astro::constants;
109 
110  if( mass < massEarth<units>())
111  {
112  return pow( mass/massEarth<units>(), boost::math::constants::third<realT>())*radEarth<units>();
113  }
114  else
115  {
116  return pow( mass/massEarth<units>(), static_cast<realT>(1)/static_cast<realT>(2.06))*radEarth<units>();
117  }

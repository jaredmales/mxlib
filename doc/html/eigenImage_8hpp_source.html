110 /// Calculate the median of an Eigen-like array.
111 /** Calculates the median of the entire array, allowing for some pixels to be ignored using a mask.
112  * Working memory can be retained between calls.
113  *
114  * \tparam imageT is an Eigen-like type
115  * \tparam maskT is an Eigen-like type
116  *
117  * \returns the median of the unmasked pixels of mat, using \ref vectorMedianInPlace().
118  *
119  * \ingroup eigen_image_processing
120  *
121  */
122 template<typename imageT, typename maskT=imageT>
123 typename imageT::Scalar imageMedian( const imageT & mat, ///< [in] the image to take the median of
124  const maskT * mask, ///< [in] if non-0, a 1/0 mask where 0 pixels are ignored.
125  std::vector<typename imageT::Scalar> * work =0 ///< [in] [optional] working memory can be retained and re-passed.
126  )
127 {
128  typename imageT::Scalar med;
129 
130  bool localWork = false;
131  if(work == 0)
132  {
133  work = new std::vector<typename imageT::Scalar>;
134  localWork = true;
135  }
136 
137  int sz = mat.size();
138 
139  if(mask)
140  {
141  sz = mask->sum();
142  }
143 
144  work->resize(sz);
145 
146  int ii = 0;
147  for(int i=0;i<mat.rows();++i)
148  {
149  for(int j=0; j<mat.cols();++j)
150  {
151  if(mask)
152  {
153  if( (*mask)(i,j) == 0) continue;
154  }
155 
156  (*work)[ii] = mat(i,j);
157  ++ii;
158  }
159  }
160 
161 
162  med = vectorMedianInPlace(*work);
163 
164  if(localWork) delete work;
165 
166  return med;
167 }

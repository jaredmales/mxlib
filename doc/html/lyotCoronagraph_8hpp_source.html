393 template<typename _realT, typename _fpmaskFloatT>
394 void lyotCoronagraph<_realT, _fpmaskFloatT>::optimizeAPLCMC( imageT & geomPupil,
395  realT fpmRadPix,
396  realT relTol,
397  realT absTol,
398  int maxIter,
399  const std::string & cname )
400 {
401  complexFieldT focalPlane, pupilPlane;
402 
403  pupilPlane.resize(_wfSz, _wfSz);
404  focalPlane.resize(_wfSz, _wfSz);
405 
406  mx::imagingArray<realT,mx::fftwAllocator<realT>, 0> mask(_wfSz, _wfSz);
407  mx::wfp::circularPupil( mask, 0., fpmRadPix);
408 
409  //Initialize pupilImage
410  mx::imagingArray<realT,mx::fftwAllocator<realT>, 0> pupilImage(_wfSz, _wfSz);
411  pupilImage.setZero();
412 
413  int gpLLi = 0.5*(_wfSz-1) - 0.5*(geomPupil.rows()-1);
414  int gpLLj = 0.5*(_wfSz-1) - 0.5*(geomPupil.cols()-1);
415 
416  int gpURi = gpLLi + geomPupil.rows();
417  int gpURj = gpLLj + geomPupil.cols();
418 
419  for(int i=0;i<geomPupil.rows();++i)
420  {
421  for(int j=0;j< geomPupil.cols();++j)
422  {
423  pupilImage(gpLLi + i, gpLLj + j) = geomPupil(i,j);
424  }
425  }
426 
427  realT lastLambdaA, LambdaA;
428 
429  lastLambdaA = 1;
430  int n;
431  std::string reason;
432  for(n=0; n< maxIter; ++n)
433  {
434  mx::wfp::makeComplexPupil(pupilPlane, pupilImage, _wfSz);
435 
436  fi.propagatePupilToFocal(focalPlane, pupilPlane);
437 
438  for(int i=0; i < _wfSz; ++i)
439  {
440  for(int j=0;j < _wfSz; ++j)
441  {
442  focalPlane(i,j) *= mask(i,j);
443  }
444  }
445 
446  fi.propagateFocalToPupil(pupilPlane, focalPlane);
447 
448  for(int i=0; i< _wfSz; ++i) for(int j=0;j< _wfSz; ++j) pupilImage(i,j) = abs(pupilPlane(i,j));
449 
450 
451  LambdaA = 0;
452  for(int i=0; i< _wfSz; ++i)
453  {
454  for(int j=0;j< _wfSz; ++j)
455  {
456  if( i >= gpLLi && i < gpURi && j >= gpLLj && j < gpURj)
457  {
458  pupilImage(i,j) *= geomPupil(i-gpLLi,j-gpLLj);
459  if(pupilImage(i,j) > LambdaA) LambdaA = pupilImage(i,j);
460  }
461  else
462  {
463  pupilImage(i,j) = 0;
464  }
465  }
466  }
467  //LambdaA = 1.0/LambdaA;
468  for(int i=0; i< _wfSz; ++i) for(int j=0;j<_wfSz; ++j) pupilImage(i,j) /= LambdaA;
469 
470  std::cout << n << " " << LambdaA << "\n";
471  if( fabs(LambdaA - lastLambdaA) < absTol)
472  {
473  std::cout << "Converged on absTol.\n";
474  reason = "absTol";
475  break;
476  }
477  if( fabs((LambdaA - lastLambdaA)/lastLambdaA) < relTol)
478  {
479  std::cout << "Converged on relTol.\n";
480  reason = "relTol";
481  break;
482  }
483 
484  if(n == maxIter - 1)
485  {
486  std::cout << "maxIter reached.\n";
487  reason = "maxIter";
488  }
489 
490  lastLambdaA = LambdaA;
491  }
492  if(LambdaA > 1) LambdaA = 1;
493 
494  std::cout << "LambdaA: = " << LambdaA << "\n";
495 
496  realT trans = 1.0 - 1.0/LambdaA;
497 
498  int pupSize = geomPupil.rows();
499 
500  pupilApodizer.resize(pupSize, pupSize);
501 
502  extractBlock(pupilApodizer, 0, pupSize, 0, pupSize, pupilImage, 0.5*( (pupilImage.rows()-1) - (pupSize-1)), 0.5*( (pupilImage.rows()-1) - (pupSize-1)));
503 
504  makeFocalMask(fpmRadPix, trans, pupSize);
505  maskSource = 2;
506 
507  lyotStop = geomPupil;
508 
509 
510  improc::fitsHeader head;
511

513  head.append("", improc::fitsCommentType(), "lyotCoronagraph optimization Parameters:");
514  head.append("", improc::fitsCommentType(), "----------------------------------------");

520  head.append<int>("NITER", n , "actual number of iterations");

522  head.append<realT>("FPMTRANS", trans, "transmission of FPM");
523
